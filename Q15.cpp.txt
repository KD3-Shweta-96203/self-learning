#include <iostream>
#include <string>
using namespace std;


class NegativeMarksException {
public:
    const char* what() const {
        return "Marks cannot be negative!";
    }
};

namespace college {

    class Student {
    private:
        int id;
        string name;
        float marks;

    public:
        // Constructor
        Student(int i, string n, float m) {
            id = i;
            name = n;
            setMarks(m); 
        }

        // Setters
        void setId(int i) {
            id = i;
        }

        void setName(string n) {
            name = n;
        }

        void setMarks(float m) {
            if (m < 0) {
                throw NegativeMarksException();
            }
            marks = m;
        }

       
        int getId() {
            return id;
        }

        string getName() {
            return name;
        }

        float getMarks() {
            return marks;
        }

      
        void display() {
            cout << "Student ID: " << id << endl;
            cout << "Name: " << name << endl;
            cout << "Marks: " << marks << endl;
        }
    };

    class Teacher {
    private:
        int id;
        string name;
        string subject;

    public:
       
        Teacher(int i, string n, string s) {
            id = i;
            name = n;
            subject = s;
        }

        // Setters
        void setId(int i) {
            id = i;
        }

        void setName(string n) {
            name = n;
        }

        void setSubject(string s) {
            subject = s;
        }

        // Getters
        int getId() {
            return id;
        }

        string getName() {
            return name;
        }

        string getSubject() {
            return subject;
        }

        // Display method
        void display() {
            cout << "Teacher ID: " << id << endl;
            cout << "Name: " << name << endl;
            cout << "Subject: " << subject << endl;
        }
    };

} 

int main() {
    try {
       
        college::Student s1(1, "Alice", 85);
        college::Teacher t1(101, "Mr. Sharma", "Mathematics");

       
        cout << "\nStudent Details:\n";
        s1.display();

        cout << "\nTeacher Details:\n";
        t1.display();

       
        cout << "\nSetting negative marks...\n";
        s1.setMarks(-10); 

    } catch (NegativeMarksException e) {
        cout << "Exception: " << e.what() << endl;
    }

    return 0;
}
