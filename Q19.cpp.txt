#include <iostream>
#include <list>
#include <algorithm>
using namespace std;

int main() {
    list<int> nums;

    
    nums = {18, 41, 9, 30, 22, 16, 7, 10, 1, 18};

    
    cout << "Reverse order: ";
    for (auto it = nums.rbegin(); it != nums.rend(); ++it) {
        cout << *it << " ";
    }
    cout << endl;


    for (auto it = nums.begin(); it != nums.end(); ++it) {
        *it += 5;
    }


    cout << "After increment (const iterator): ";
    for (auto it = nums.cbegin(); it != nums.cend(); ++it) {
        cout << *it << " ";
    }
    cout << endl;


    nums.sort();


    cout << "Sorted list: ";
    for (auto it = nums.begin(); it != nums.end(); ++it) {
        cout << *it << " ";
    }
    cout << endl;

    return 0;
}
